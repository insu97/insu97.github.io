```html
<!doctype html>
<html lang = "ko">

    <head>
        <meta charset="UTF-8">
        <title>Latex</title>

        <!-- 라텍스 사용하려면 이거 입력!!-->
        <script type="text/x-mathjax-config">
            MathJax.Hub.Config({
                tex2jax: {inlineMath: [['$','$'], ['\\(','\\)']]}
            });
        </script>

        <script src='https://cdnjs.cloudflare.com/ajax/libs/mathjax/2.7.5/latest.js?config=TeX-MML-AM_CHTML' async>		   </script>
        <!----------->
    </head>

    <body>
        <!-- b : bold , 글자 진하게 ( 단순 ) -->
        <b>글자 진함</b>

        <!-- strong : 글자 굵게 ( 중요 의미 ) -->
        <strong>글자 중요함</strong>

        <!-- i : italic , 글자 기울이기 -->
        <i>글자가 기울어짐</i>

        <!-- em : emphasize , 글자 강조 -->
        <em>글자를 강조하고 싶다면 사용</em>

        <!-- mark : 형광펜 칠하기 -->
        <mark>형광</mark>

        <!-- del : 취소선 긋기 -->
        <del>취소선입니다</del>

        <!-- ins : insert , 글자 채우는 표현 -->
        <ins>글자 채우기</ins>


    </body>

</html>
```
