```html
<!doctype html>
<html lang = "ko">

    <head>
        <meta charset="UTF-8">
        <title>링크</title>
    </head>

    <body>
        <!-- a : 앵커 , 절대경로 지정방식으로 문처 참조 -->
        <a href = "https://naver.com">클릭 시 네이버로 이동</a>

        <a href = "#bookmark">특정 위치로 이동시키고 싶을때</a>

        <p name = "bookname">
            여기로 이동 됩니다!
        </p>
    </body>
</html>
```
