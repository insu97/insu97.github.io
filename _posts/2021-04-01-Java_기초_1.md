```java
package STUDY;

public class Variable_ {

    // 변수명은 알파벳(대/소문자) , 숫자 , 특수문자(_,$) , 한글 사용 가능
    // 숫자로 시작 불가능
    // 공백 사용 불가
    // 예약어 사용 불가

    // 클래스명은 첫자를 대문자!
    // Camel 표기법 : lastIndexOf , parseInt ...
    // snake 표기법 : student_number
    // 상수는 모두 대문자로 : PI

    public static void main(String[] args) {
        
        int a;      // 변수 선언
        a = 5;      // 변수 초기화

        int b = 5 ; // 위 과정을 한번에!
        
    }

}
```
