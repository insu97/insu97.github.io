```html
<!doctype html>
<html lang = "ko">

    <head>
        <meta charset="UTF-8">
        <title>본문 표현</title>
    </head>

    <body>
        <!-- 제목 -->
        <h1>제목1</h1>
        <h2>제목2</h2>
        <h3>제목3</h3>
        <h4>제목4</h4>
        <h5>제목5</h5>

        <!-- 줄긋기 <hr> 이나 <hr/> 둘다 사용 가능-->
        <hr/>

        <!-- 단락 -->
        <p>
            안녕하세요 .
        </p>

        <!-- 줄바꿈 -->
        <br/>

        <!-- 인용구 -->
        <q>인용구 사용</q>	<!-- 인용구의 길이가 짧을 때 -->

        <blockquote>
            <p>
                인용구 사용	<!-- 인용구의 길이가 길 때 -->
            </p>
        </blockquote>

    </body>

</html>
```
