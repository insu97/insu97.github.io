> 조인 검색

```mysql
# 기본식
select *
from 테이블명1 , 테이블명2;  # ( 카테시안 곱 )
```

```mysql
# 조인 후 필터링
SELECT *
FROM 테이블명1 as "A", 테이블명2 as "B"
where 조건식 작성 ;
```

```mysql
# 최신식 (= inner join)
SELECT *
FROM 테이블명1 inner join 테이블명2
using (공통된 컬럼명)
where 조건식 작성;
```

```mysql
# 중첩
SELECT *
from 테이블명1 
inner join 테이블명2 using(컬럼명1) 
inner join 테이블명3 using(컬럼명2)
where 조건식;
```

```mysql
# left outer join
SELECT *
from 테이블명1 left outer join 테이블명2
using(컬럼명);
#--------------------------------------
SELECT *
from 테이블명1 left outer join 테이블명2
using(컬럼명)
where 조건식;
```

```mysql
# right outer join
SELECT *
from 테이블명1 right outer join 테이블명2
using(컬럼명);
#--------------------------------------
SELECT *
from 테이블명1 right outer join 테이블명2
using(컬럼명)
where 조건식;
```

```mysql
# self 조인 : 같은 테이블 2개를 이용해서 결합하는 조인
# inner join , outer join 상관 없음
# 계층구조일 때 사용
SELECT *
from 테이블명1 inner join 테이블명1
on 테이블명1.컬럼명1 = 테이블명1.컬럼명1
order by 테이블명1.컬럼명1 ;
```

```mysql
# view
# 테이블과 동일한 속성을 가진다
create view 뷰이름
as select *
from 테이블명
where 조건식 ;
#----------------
drop view 우수고객;
```
