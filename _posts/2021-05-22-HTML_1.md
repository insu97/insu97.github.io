```html
<!-- 주석 다는 방법 -->

<!-- 기본으로 작성해야 하는 HTML 코드 -->
<!doctype html>
<html lang = "ko">
    
    <head>
        <meta charset="UTF-8">
        <title>타이틀 제목을 입력하는 공간</title>
    </head>
    
    <body>
        <!-- 내용 부분 입력 -->
    </body>
    
</html>
```

```html
<!doctype html>
<html lang = "ko">
    
    <head>
        <meta charset="UTF-8">
        <title>본문 표현</title>
    </head>
    
    <body>
        <!-- 제목 -->
        <h1>제목1</h1>
        <h2>제목2</h2>
        <h3>제목3</h3>
        <h4>제목4</h4>
        <h5>제목5</h5>
        
        <!-- 줄긋기 <hr> 이나 <hr/> 둘다 사용 가능-->
        <hr/>
        
        <!-- 단락 -->
        <p>
            안녕하세요 .
        </p>
        
        <!-- 줄바꿈 -->
        <br/>
        
    </body>
    
</html>
```
